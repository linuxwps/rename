interface SequentialConfig {
  enabled: boolean;
  startAt: number;       // default: 1
  digits: number;        // default: 3
  position: "before" | "after";  // default: "after"
}

interface RegexConfig {
  enabled: boolean;
  pattern: string;
  replacement: string;
  caseSensitive: boolean;  // default: false
}

interface PrefixConfig {
  enabled: boolean;
  text: string;
}

interface SuffixConfig {
  enabled: boolean;
  text: string;
}

interface ExtensionConfig {
  enabled: boolean;
  mode: "change" | "remove" | "add";  // default: "change"
  newExtension: string;
}

interface ReplaceConfig {
  enabled: boolean;
  findText: string;
  replaceText: string;
  caseSensitive: boolean;  // default: false
}

interface RenameModeStates {
  sequential: SequentialConfig;
  regex: RegexConfig;
  prefix: PrefixConfig;
  suffix: SuffixConfig;
  extension: ExtensionConfig;
  replace: ReplaceConfig;
}

interface DiffSegment {
  value: string;
  type: "added" | "removed" | "unchanged";
}

interface PreviewResult {
  fileId: string;
  oldBaseName: string;
  oldExtension: string;
  newBaseName: string;
  newExtension: string;
  newFullName: string;
  diffBaseName: DiffSegment[];
  diffExtension: DiffSegment[];
  hasConflict: boolean;
  conflictWith: string[];
}

    - import FileItem from types/file.ts
    - import 6 个 config 接口 + RenameModeStates from types/rename.ts
    - 导出常量 MODE_ORDER: Array&lt;keyof RenameModeStates&gt; = ["sequential", "regex", "replace", "prefix", "suffix", "extension"]（严格按 D-14 顺序）
    - 导出一个纯函数 `applyRenamePipeline(file: FileItem, modes: RenameModeStates, index: number, totalFiles: number): { newBaseName: string; newExtension: string }`

    applyRenamePipeline 内部逻辑：
    a) 从 file.name 中分离 baseName（去掉扩展名部分: file.name.replace(`.${file.extension}`, "")）和 newExtension（初始值 = file.extension）
    b) 遍历 MODE_ORDER 中的每个 mode
    c) 如果 modes[mode].enabled 为 false，跳过
    d) 按 mode 类型执行转换：
       - sequential: 根据 startAt + index 计算序号（`(startAt + index).toString().padStart(digits, "0")`），按 position 决定 "before"（插前面）或 "after"（插后面加 `_` 分隔符）
       - regex: `new RegExp(pattern, caseSensitive ? "g" : "gi")` + `baseName.replace(regex, replacement)`。必须用 try-catch 包裹，捕获异常时跳过（不修改 baseName）
       - replace: 对 findText 做正则转义（`.replace(/[.*+?^${}()|[\]\\]/g, "\\$&")`），构建 RegExp(caseSensitive ? "g" : "gi")，执行 replace
       - prefix: baseName = text + baseName
       - suffix: baseName = baseName + text
       - extension: mode === "change" && newExtension 非空 → 更新 newExtension；mode === "remove" → newExtension = ""；mode === "add" && 原无扩展名 → 更新 newExtension
    e) 返回 { newBaseName, newExtension }

    所有函数必须纯函数，无副作用，无 React 依赖。添加 JSDoc 注释说明每个步骤的作用。
  </action>

    导入：
    - useMemo from "react"
    - diffWords from "diff"
    - FileItem from types/file.ts
    - RenameModeStates, PreviewResult, DiffSegment from types/rename.ts
    - applyRenamePipeline from utils/renamePipeline.ts

    hook 签名：`export function useRenameEngine(files: FileItem[], modes: RenameModeStates): { previews: PreviewResult[]; totalConflicts: number }`

    内部逻辑（全部在 useMemo 内，deps: [files, modes]）：

    a) 对 files 做浅拷贝并按 modifiedAt 升序排序（副本排序，不修改原数组）。排序键: `(a.modifiedAt ?? 0) - (b.modifiedAt ?? 0)`

    b) 对于每个排序后的文件，调用 `applyRenamePipeline(file, modes, index, files.length)` 获取 { newBaseName, newExtension }
    
    c) 组装 newFullName: newExtension ? `${newBaseName}.${newExtension}` : newBaseName

    d) 调用 `diffWords(oldBaseName, newBaseName)` 获取 baseName 部分的 diff 分段。将 diff 结果映射为 DiffSegment[]，其中 `part.added` → type "added", `part.removed` → type "removed", 否则 type "unchanged"

    e) 对 extension: 如果 oldExtension !== newExtension，调用 diffWords 对比（加 "." 前缀）；否则返回单一 unchanged 段

    f) 组装 PreviewResult 对象（fileId, oldBaseName, oldExtension, newBaseName, newExtension, newFullName, diffBaseName, diffExtension, hasConflict: false, conflictWith: []）

    g) 执行 detectConflicts(previews)：使用 Map&lt;string, number[]&gt;，key 为 newFullName.toLowerCase()（APFS 大小写不敏感，per RESEARCH.md 避免冲突漏检）。遍历 previews 填充 Map。对每个 value.length > 1 的 bucket，将该 bucket 内所有 index 的 previews[i].hasConflict 设为 true，conflictWith 设为冲突文件名列表

    h) 计算 totalConflicts = previews.filter(p => p.hasConflict).length

    i) 返回 { previews, totalConflicts }

    所有 try-catch 包裹的 RegExp 创建在 renamePipeline.ts 中已处理，此处不需要额外 try-catch。

    JSDoc 注释说明 hook 的职责和参数。
